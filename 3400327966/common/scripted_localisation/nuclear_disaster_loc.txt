# 核灾难与战争伤亡统计的 scripted localisation。
# 每个定义都根据显示变量的数量级切换文本键，分别输出普通数值、K、M、KM 等格式。
# 带 selected_casualty_country 的定义读取事件目标国家变量；global.* 则直接读取全局变量。
custom_localization = {
	name = GetGlobalPeopleDeathByNukesText

	text = {
		trigger = {
			check_variable = { global.people_death_by_nukes_display > 1000000000 }
		}
		localization_key = global_people_death_by_nukes_text_km
	}
	text = {
		trigger = {
			check_variable = { global.people_death_by_nukes_display > 1000000 }
		}
		localization_key = global_people_death_by_nukes_text_m
	}
	text = {
		trigger = {
			check_variable = { global.people_death_by_nukes_display > 1000 }
		}
		localization_key = global_people_death_by_nukes_text_k
	}
	text = {
		localization_key = global_people_death_by_nukes_text
	}
}
