# 重建状态优先级：可完成 > 被锁定 > 暂停 > 施工中 > 等待中。
custom_localization = {
	name = GetWastelandRebuildStatusText

	text = {
		trigger = { check_variable = { wasteland_rebuild_ready > 0 } }
		localization_key = wasteland_rebuild_status_ready
	}
	text = {
		trigger = { is_demilitarized_zone = yes }
		localization_key = wasteland_rebuild_status_locked
	}
	text = {
		trigger = { check_variable = { wasteland_rebuild_stop_days > 0 } }
		localization_key = wasteland_rebuild_status_stopped
	}
	text = {
		trigger = { check_variable = { wasteland_rebuild_stage > 0 } }
		localization_key = wasteland_rebuild_status_working
	}
	text = {
		localization_key = wasteland_rebuild_status_waiting
	}
}

# 完成时间文本目前只区分“可立即完成”和“尚未解锁/不可完成”两类。
custom_localization = {
	name = GetWastelandRebuildFinishText

	text = {
		trigger = {
			check_variable = { wasteland_rebuild_ready > 0 }
			NOT = { is_owned_by = ROOT }
		}
		localization_key = wasteland_rebuild_finish_not_owner
	}
	text = {
		trigger = { check_variable = { wasteland_rebuild_ready > 0 } }
		localization_key = wasteland_rebuild_finish_ready
	}
	text = {
		localization_key = wasteland_rebuild_finish_locked
	}
}
