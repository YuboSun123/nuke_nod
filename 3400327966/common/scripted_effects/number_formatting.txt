#--------------- 全局难民数字单位换算（K/M/KM） ---------------#
refresh_refugee_state_display_variables = {
	# 州级派生变量只依赖本州数据，可在州作用域下单独刷新，避免日频全图重算。
	format_state_refugee_pool_display = yes

	set_variable = { var = state_refugee_ratio_pct value = 0 }
	if = {
		limit = { state_population_k > 0 }
		set_temp_variable = { var = refugee_ratio_temp value = state_refugee_pool_k }
		divide_temp_variable = { var = refugee_ratio_temp value = state_population_k }
		multiply_temp_variable = { var = refugee_ratio_temp value = 100 }
		set_variable = { var = state_refugee_ratio_pct value = refugee_ratio_temp }
	}

	set_variable = { var = state_refugee_resettled_ratio_pct value = 0 }
	if = {
		limit = {
			OR = {
				check_variable = { state_refugee_resettled_total_people > 0 }
				check_variable = { state_refugee_pool_k > 0.001 }
			}
		}
		set_temp_variable = { var = refugee_resettled_total_flow_people value = state_refugee_pool_k }
		multiply_temp_variable = { var = refugee_resettled_total_flow_people value = 1000 }
		add_to_temp_variable = { var = refugee_resettled_total_flow_people value = state_refugee_resettled_total_people }
		if = {
			limit = { check_variable = { var = refugee_resettled_total_flow_people value = 0 compare = greater_than } }
			set_temp_variable = { var = refugee_resettled_ratio_temp value = state_refugee_resettled_total_people }
			divide_temp_variable = { var = refugee_resettled_ratio_temp value = refugee_resettled_total_flow_people }
			multiply_temp_variable = { var = refugee_resettled_ratio_temp value = 100 }
			set_variable = { var = state_refugee_resettled_ratio_pct value = refugee_resettled_ratio_temp }
		}
	}

	format_state_refugee_weekly_pop_delta_display = yes
	update_homeless_work_disruption_drive_values_for_state = yes
}

update_refugee_display_variables = { 
	# 国家/全局难民池按州难民池重算，避免控制权变化和长期增量更新带来的统计漂移。
	
	set_variable = { var = global.refugee_pool_k value = 0.0 }
	every_country = {
		set_variable = { var = country_refugee_pool_k value = 0.0 }
	}
	every_state = {
		if = {
			limit = { check_variable = { state_refugee_pool_k > 0 } }
			controller = {
				add_to_variable = { var = country_refugee_pool_k value = PREV.state_refugee_pool_k }
			}
			add_to_variable = { var = global.refugee_pool_k value = state_refugee_pool_k }
		}
	}
	
	every_state = {
		refresh_refugee_state_display_variables = yes
	}

	every_country = { 
		# 州难民占国家难民人口比（%）：州难民池 / 当前控制国难民池 * 100
		every_controlled_state = {
			set_variable = { var = state_refugee_country_ratio_pct value = 0 }
			if = {
				limit = {
					check_variable = { state_refugee_pool_k > 0 }
					check_variable = { PREV.country_refugee_pool_k > 0 }
				}
				set_temp_variable = { var = refugee_country_ratio_temp value = state_refugee_pool_k }
				divide_temp_variable = { var = refugee_country_ratio_temp value = PREV.country_refugee_pool_k }
				multiply_temp_variable = { var = refugee_country_ratio_temp value = 100 }
				set_variable = { var = state_refugee_country_ratio_pct value = refugee_country_ratio_temp }
			}
		}

		# 国家难民池人数（功能变量）：country_refugee_pool_k（单位：K，千人）
		# 这里仅生成显示用变量，不要改写 country_refugee_pool_k 本身。
		# 真实人数（人）：用于 <1K 时显示
		format_country_refugee_pool_display = yes

		# 国家难民出入境人口差（功能变量）：country_refugee_border_net_weekly_k（单位：K，千人）
		# 仅生成显示用变量：
		# - country_refugee_border_net_weekly_people：<1K 时显示真实人数
		# - country_refugee_border_net_weekly_m / _km：大数单位
		format_country_refugee_border_net_weekly_display = yes
	}

	# 全局难民池人数（功能变量）：global.refugee_pool_k（单位：K，千人）
	# 这里仅生成显示用变量，不要改写 global.refugee_pool_k 本身。
	# 真实人数（人）：用于 <1K 时显示
	format_global_refugee_pool_display = yes
}

	set_temp_variable = { var = refugee_temp value = state_refugee_weekly_pop_delta }
	set_variable      = { var = state_refugee_weekly_pop_delta_display value = refugee_temp }

	set_temp_variable = { var = refugee_temp value = state_refugee_weekly_pop_delta_display }
	multiply_temp_variable = { var = refugee_temp value = 0.001 }
	set_variable      = { var = state_refugee_weekly_pop_delta_k value = refugee_temp }
	set_temp_variable = { var = refugee_temp value = state_refugee_weekly_pop_delta_display }
	multiply_temp_variable = { var = refugee_temp value = 0.001 }
	multiply_temp_variable = { var = refugee_temp value = 0.001 }
	set_variable      = { var = state_refugee_weekly_pop_delta_m value = refugee_temp }
	set_temp_variable = { var = refugee_temp value = state_refugee_weekly_pop_delta_display }
	multiply_temp_variable = { var = refugee_temp value = 0.001 }
	multiply_temp_variable = { var = refugee_temp value = 0.001 }
	multiply_temp_variable = { var = refugee_temp value = 0.001 }
	set_variable      = { var = state_refugee_weekly_pop_delta_km value = refugee_temp }
}
