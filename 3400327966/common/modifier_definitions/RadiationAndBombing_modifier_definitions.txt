#	# tooltip 显示精度（小数位数）
#	示例：precision = 2
#	
#	# tooltip 值后缀

#	示例：postfix = none

#	# none all naval air unit_leader country state army peace politics ai defensive aggressive

#	# 默认是 all（包含所有类别）
